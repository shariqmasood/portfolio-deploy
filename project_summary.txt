import os
import smtplib

from dotenv import load_dotenv

    return render_template("splash.html")

@app.route("/index")
def home():
    return render_template("index.html")

@app.route("/about")
def about():
    return render_template("about.html")

@app.route("/resume")
def resume():
    return render_template("resume.html")

@app.context_processor
def inject_current_year():
    return {'current_year': datetime.now().year}

def contact():
    if request.method == "POST":
        data = request.form

def send_email(name, sender_email, phone, message):
    # Email to yourself
    # Email to yourself

    Phone: {phone}

# === export_project.py ===

import os

output_file = "project_summary.txt"

# Directories and extensions to skip
exclude_dirs = {'.git', '.venv', '__pycache__', '.idea', 'node_modules', 'dist', 'build'}
exclude_exts = {
    '.png', '.jpg', '.jpeg', '.gif', '.ico', '.exe', '.dll', '.pyd', '.pyc',
    '.so', '.zip', '.tar', '.gz', '.pdf', '.db', '.woff', '.woff2', '.ttf'
}

def should_skip(path):
    return any(part in exclude_dirs for part in path.split(os.sep)) or \
           os.path.splitext(path)[1].lower() in exclude_exts

def read_file_safe(file_path):
    try:
        with open(file_path, "r", encoding="utf-8") as f:
            return f.read()
    except Exception as e:
        return f"Error reading {file_path}: {e}\n"

summary = []

for root, dirs, files in os.walk(".", topdown=True):
    dirs[:] = [d for d in dirs if d not in exclude_dirs]
    for file in files:
        rel_path = os.path.relpath(os.path.join(root, file), start=".")
        if should_skip(rel_path):
            continue
        content = read_file_safe(os.path.join(root, file))
        summary.append(f"\n\n# === {rel_path} ===\n\n{content}")

with open(output_file, "w", encoding="utf-8") as f:
    f.writelines(summary)

print("✅ Clean summary created as 'project_summary.txt' without .git and binary errors.")

### FILE: export_project.py ###
import os

# Define file extensions you want to include
valid_extensions = ['.py', '.html', '.css', '.js', '.txt', '.md', '', '.json']

# Define the output file
output_file = "project_summary.txt"

# Walk through all directories and files
with open(output_file, "w", encoding='utf-8') as out:
    for root, dirs, files in os.walk("."):
        for file in files:
            ext = os.path.splitext(file)[1]
            if ext in valid_extensions and ".venv" not in root and "__pycache__" not in root and ".idea" not in root:
                full_path = os.path.join(root, file)
                try:
                    with open(full_path, "r", encoding='utf-8') as f:
                        rel_path = os.path.relpath(full_path, ".")
                        out.write(f"\n\n### FILE: {rel_path} ###\n")
                        out.write(f.read())
                        out.write("\n\n" + "#" * 60 + "\n")
                except Exception as e:
                    print(f"Error reading {full_path}: {e}")

print(f"\n✅ Project exported to: {output_file}")

<script>
  document.addEventListener("DOMContentLoaded", function () {
    const navLinks = document.querySelectorAll('.nav-link-scroll');
    const sections = [...document.querySelectorAll('h2[id]')];
    const floatingNav = document.getElementById('floatingNav');
    const toggleNav = document.getElementById('toggleNav');

    let lastScrollTop = 0;
    let ticking = false;

    function onScroll() {
      const scrollPosition = window.scrollY + 140;
      let current = sections.find(sec => sec.offsetTop <= scrollPosition);
      navLinks.forEach(link => link.classList.remove('active'));
      if (current) {
        const activeLink = document.querySelector(`.nav-link-scroll[href="#${current.id}"]`);
        if (activeLink) activeLink.classList.add('active');
      }

      const currentScrollTop = window.scrollY;
      if (!ticking) {
        window.requestAnimationFrame(() => {
          if (currentScrollTop > lastScrollTop && window.innerWidth >= 768) {
            floatingNav.classList.add('fade-out');
            floatingNav.classList.remove('fade-in');
          } else {
            floatingNav.classList.add('fade-in');
            floatingNav.classList.remove('fade-out');
          }
          lastScrollTop = currentScrollTop <= 0 ? 0 : currentScrollTop;
          ticking = false;
        });
        ticking = true;
      }
    }

    document.addEventListener("scroll", onScroll);
    onScroll();

    // Toggle nav visibility on mobile
    toggleNav.addEventListener("click", function () {
      floatingNav.classList.toggle("d-none");
    });
  });
</script>

  document.addEventListener("DOMContentLoaded", function () {
    const navLinks = document.querySelectorAll('.nav-link-scroll');
    const sections = [...document.querySelectorAll('h2[id]')];
    const floatingNav = document.getElementById('floatingNav');
    const toggleNav = document.getElementById('toggleNav');

    let lastScrollTop = 0;
    let ticking = false;

    function onScroll() {
      const scrollPosition = window.scrollY + 140;
      let current = sections.find(sec => sec.offsetTop <= scrollPosition);
      navLinks.forEach(link => link.classList.remove('active'));
      if (current) {
        const activeLink = document.querySelector(`.nav-link-scroll[href="#${current.id}"]`);
        if (activeLink) activeLink.classList.add('active');
      }

      const currentScrollTop = window.scrollY;
      if (!ticking) {
        window.requestAnimationFrame(() => {
          if (currentScrollTop > lastScrollTop && window.innerWidth >= 768) {
            floatingNav.classList.add('fade-out');
            floatingNav.classList.remove('fade-in');
          } else {
            floatingNav.classList.add('fade-in');
            floatingNav.classList.remove('fade-out');
          }
          lastScrollTop = currentScrollTop <= 0 ? 0 : currentScrollTop;
          ticking = false;
        });
        ticking = true;
      }
    }

    document.addEventListener("scroll", onScroll);
    onScroll();

    // Toggle nav visibility on mobile
    toggleNav.addEventListener("click", function () {
      floatingNav.classList.toggle("d-none");
    });
  });
</script>

<script>
  document.addEventListener("DOMContentLoaded", function () {
    const navLinks = document.querySelectorAll('.nav-link-scroll');
    const sections = [...document.querySelectorAll('h2[id]')];
    const floatingNav = document.getElementById('floatingNav');
    const toggleNav = document.getElementById('toggleNav');

    let lastScrollTop = 0;
    let ticking = false;

    function onScroll() {
      const scrollPosition = window.scrollY + 140;
      let current = sections.find(sec => sec.offsetTop <= scrollPosition);
      navLinks.forEach(link => link.classList.remove('active'));
      if (current) {
        const activeLink = document.querySelector(`.nav-link-scroll[href="#${current.id}"]`);
        if (activeLink) activeLink.classList.add('active');
      }

      const currentScrollTop = window.scrollY;
      if (!ticking) {
        window.requestAnimationFrame(() => {
          if (currentScrollTop > lastScrollTop && window.innerWidth >= 768) {
            floatingNav.classList.add('fade-out');
            floatingNav.classList.remove('fade-in');
          } else {
            floatingNav.classList.add('fade-in');
            floatingNav.classList.remove('fade-out');
          }
          lastScrollTop = currentScrollTop <= 0 ? 0 : currentScrollTop;
          ticking = false;
        });
        ticking = true;
      }
    }

    document.addEventListener("scroll", onScroll);
    onScroll();

    // Toggle nav visibility on mobile
    toggleNav.addEventListener("click", function () {
      floatingNav.classList.toggle("d-none");
    });
  });
</script>
